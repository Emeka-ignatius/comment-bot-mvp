# Comment Bot MVP - Environment Variables Template
# Copy this file to .env in your local project directory

# ============================================
# REQUIRED: Database Configuration
# ============================================
DATABASE_URL=your_database_url_here
# Example: mysql://user:password@localhost:3306/comment_bot

# ============================================
# REQUIRED: JWT Secret
# ============================================
JWT_SECRET=your_jwt_secret_here
# Generate a random string (32+ characters recommended)
# Example: openssl rand -base64 32

# ============================================
# REQUIRED: OpenAI API (Direct, not via Manus)
# ============================================
OPENAI_API_KEY=your_openai_api_key_here
OPEN_AI_BASE_URL=https://api.openai.com
# Get your API key from: https://platform.openai.com/api-keys

# ============================================
# REQUIRED: Local Development Mode
# ============================================
LOCAL_DEV_MODE=true
# Set to 'true' to bypass authentication and use mock user

# ============================================
# OPTIONAL: Backend URL
# ============================================
VITE_BACKEND_URL=http://localhost:3000

# ============================================
# OPTIONAL: Analytics
# ============================================
VITE_ANALYTICS_ENDPOINT=
VITE_ANALYTICS_WEBSITE_ID=

# ============================================
# OPTIONAL: App Configuration
# ============================================
VITE_APP_TITLE=Comment Bot MVP
VITE_APP_LOGO=

# ============================================
# REQUIRED: Job Queue Configuration
# ============================================
MOCK_AUTOMATION=false
# Set to 'true' for testing without real API calls

# ============================================
# OPTIONAL: Puppeteer Configuration
# ============================================
PUPPETEER_EXECUTABLE_PATH=
# Leave empty for auto-detection
# Or set to custom Chrome path if needed

# ============================================
# OPTIONAL: Google Maps API (if using maps feature)
# ============================================
GOOGLE_MAPS_API_KEY=
# Only needed if using Google Maps features

# ============================================
# LOCAL DEVELOPMENT NOTES:
# ============================================
# 1. Copy this file to .env in project root
# 2. Fill in all REQUIRED values
# 3. Run: pnpm install (to remove Manus dependencies)
# 4. Run: pnpm exec playwright install chromium
# 5. Run: pnpm db:push (to sync database schema)
# 6. Run: pnpm dev
# 7. Open: http://localhost:3000
